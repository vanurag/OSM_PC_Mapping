97  {
98  virtual Interface* createInstance(const std::string& className, const Parametrizable::Parameters& params) const
99  {
100  C* instance(new C(params));
101 
102  // check that there was no unsed parameter
103  for (BOOST_AUTO(it, params.begin()); it != params.end() ;++it)
104  {
105  if (instance->parametersUsed.find(it->first) == instance->parametersUsed.end())
106  throw Parametrizable::InvalidParameter(
107  (boost::format("Parameter %1% for module %2% was set but is not used") % it->first % className).str()
108  );
109  }
110 
111  return instance;
112  }
